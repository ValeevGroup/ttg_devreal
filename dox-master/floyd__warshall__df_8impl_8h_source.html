33 struct Key {
34  // ((I, J), K) where (I, J) is the tile coordiante and K is the iteration number
35  std::pair<std::pair<int, int>, int> execution_info;
36 
37  bool operator==(const Key& b) const {
38  if (this == &b) return true;
39  return execution_info.first.first == b.execution_info.first.first &&
40  execution_info.first.second == b.execution_info.first.second &&
41  execution_info.second == b.execution_info.second;
42  }
43 
44  bool operator!=(const Key& b) const { return !((*this) == b); }
45 
46  madness::hashT hash_val;
47 
48  Key() : execution_info(std::make_pair(std::make_pair(0, 0), 0)) { rehash(); }
49  Key(const std::pair<std::pair<int, int>, int>& e) : execution_info(e) { rehash(); }
50  Key(int e_f_f, int e_f_s, int e_s) : execution_info(std::make_pair(std::make_pair(e_f_f, e_f_s), e_s)) { rehash(); }
51 
52  madness::hashT hash() const { return hash_val; }
53  void rehash() {
54  std::hash<int> int_hasher;
55  hash_val = int_hasher(execution_info.first.first) ^ int_hasher(execution_info.first.second) ^
56  int_hasher(execution_info.second);
57  }

112  void op(const int& iterations, typename baseT::output_terminals_type& out) {
113  // making x_ready for all the blocks (for function calls A, B, C, and D)
114  // This triggers for the immediate execution of function A at tile [0, 0]. But
115  // functions B, C, and D have other dependencies to meet before execution; They wait
116 #ifdef HAS_EXECUTION_HEADER
117  std::for_each(std::execution::par, adjacency_matrix_ttg->get().begin(), adjacency_matrix_ttg->get().end(),
118  [&out](const std::pair<std::pair<int,int>, BlockMatrix<T>>& kv)
119 #else
120  std::for_each(adjacency_matrix_ttg->get().begin(), adjacency_matrix_ttg->get().end(),
121  [&out](const std::pair<std::pair<int,int>, BlockMatrix<T>>& kv)
122 #endif
123  {
124  auto [i,j] = kv.first;
125  if (i == 0 && j == 0) { // A function call
126  ::send<0>(Key(std::make_pair(std::make_pair(i, j), 0)), kv.second, out);
127  } else if (i == 0) { // B function call
128  ::send<1>(Key(std::make_pair(std::make_pair(i, j), 0)), kv.second, out);
129  } else if (j == 0) { // C function call
130  ::send<2>(Key(std::make_pair(std::make_pair(i, j), 0)), kv.second, out);
131  } else { // D function call
132  ::send<3>(Key(std::make_pair(std::make_pair(i, j), 0)), kv.second, out);
133  }
134  });
