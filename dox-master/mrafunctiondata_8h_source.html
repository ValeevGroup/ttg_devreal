23  static void make_abgv_diff_operator() {
24  double iphase = 1.0;
25  for (auto i: range(K)) {
26  double jphase = 1.0;
27  for (auto j : range(K)) {
28  double gammaij = std::sqrt(double((2*i+1)*(2*j+1)));
29  double Kij;
30  if (((i-j)>0) && (((i-j)%2)==1))
31  Kij = 2.0;
32  else
33  Kij = 0.0;
34 
35  r0(i,j) = T(0.5*(1.0 - iphase*jphase - 2.0*Kij)*gammaij);
36  rm(i,j) = T(0.5*jphase*gammaij);
37  rp(i,j) = T(-0.5*iphase*gammaij);
38  }
39  }
