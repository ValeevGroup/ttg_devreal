18  class Slice {
19  public:
20  static constexpr long END = std::numeric_limits<long>::max();
21  long start; //< Start of slice (must be signed type)
22  long finish; //< Inclusive end of slice (must be signed type)
23  long step; //< Stride for slice (must be signed type)
24  long count; //< Number of elements in slice (not known until dimension is applied; negative indicates not computed)
25  Slice() : start(0), finish(END), step(1), count(-1) {}; // indicates entire range
26  Slice(long start) : start(start), finish(start+1), step(1) {} // a single element
27  Slice(long start, long end, long step=1) : start(start), finish(end), step(step) {};

30  void apply_dim(long dim) {
31  if (start == END) {start = dim-1;}
32  else if (start < 0) {start += dim;}
33 
34  if (finish == END && step > 0) {finish = dim;}
35  else if (finish == END && step < 0) {finish = -1;}
36  else if (finish < 0) {finish += dim;}
37 
38  count = std::max(0l,((finish-start-step/std::abs(step))/step+1));
39  assert((count==0) || ((count<=dim) && (start>=0 && start<=dim)));
40  finish = start + count*step; // finish is one past the last element
41  }

143  template <typename tensorT, typename opT>
144  void apply_unaryop(tensorT& t, const opT& op) {
145  if constexpr (t.is_contiguous()) { // by definition not a slice tensor
146  std::for_each(t.data().begin(), t.data().end(), op);
147  }
148  else if (t.stride(t.ndim()-1) == 1) { // inner loop is contiguous
149  if constexpr (t.ndim() == 1) {
150  const size_t n = t.dim(t.ndim()-1);
151  auto p = t.ptr();
152  for (size_t i=0; i<n; ++i) op(p[i]);
153  }
154  else if constexpr (t.ndim() >= 2) {
155  const size_t dimi = t.dim(t.ndim()-1);
156  const size_t dimj = t.dim(t.ndim()-2);
157  const size_t stridei = t.stride(t.ndim()-1);
158  const size_t stridej = t.stride(t.ndim()-2);
159  if (stridei == 1) {
160  for (auto it=t.xbegin(); it!=t.xend(); ++it) {
161  auto p = t.ptr(it);
162  for (size_t j=0; j<dimj; ++j, p+=stridej) {
163  for (size_t i=0; i<dimi; ++i) {
164  op(p[i]);
165  }
166  }
167  }
168  }
169  else {
170  for (auto it=t.xbegin(); it!=t.xend(); ++it) {
171  auto p = t.ptr(it);
172  for (size_t j=0; j<dimj; ++j, p+=stridej) {
173  auto q = p;
174  for (size_t i=0; i<dimi; ++i, q+=stridei) {
175  op(*q);
176  }
177  }
178  }
179  }
180  }
181  }
182  else { // general but slowest iteration
183  std::cout << "in slow apply_unaryop ... what are you doing?\n";
184  std::for_each(t.begin(), t.end(), op);
185  }
186  }

190  void apply_binaryop(tensorA& a, tensorB& b, const opT& op) {
191  assert(a.conforms(b));
192  if (a.is_contiguous() && b.is_contiguous()) { // by definition not a slice tensor
193  std::cout << "bina ... not been here yet?\n";
194  auto pa = a.ptr();
195  auto pb = b.ptr();
196  for (size_t i=0; i<a.size(); i++) op(pa[i],pb[i]);
197  }
198  else if (a.stride(a.ndim()-1) == 1 && b.stride(b.ndim()-1) == 1) { // inner loop is contiguous
199  if (a.ndim() == 1) {
200  std::cout << "binb ... not been here yet?\n";
201  const size_t n = a.dim(a.ndim()-1);
202  auto pa = a.ptr();
203  auto pb = b.ptr();
204  for (size_t i=0; i<n; ++i) op(pa[i],pb[i]);
205  }
206  else if (a.ndim() >= 2) {
207  const size_t dimi = a.dim(a.ndim()-1);
208  const size_t dimj = a.dim(a.ndim()-2);
209  const size_t strideia = a.stride(a.ndim()-1);
210  const size_t strideja = a.stride(a.ndim()-2);
211  const size_t strideib = b.stride(b.ndim()-1);
212  const size_t stridejb = b.stride(b.ndim()-2);
213  auto ita = a.xbegin();
214  auto itb = b.xbegin();
215  const auto enda = a.xend();
216  const auto endb = b.xend();
217  if (strideia == 1 && strideib == 1) {
218  //std::cout << "binc\n";
219  while (ita != enda) {
220  auto pa = a.ptr(ita);
221  auto pb = b.ptr(itb);
222  for (size_t j=0; j<dimj; ++j, pa+=strideja, pb+=stridejb) {
223  for (size_t i=0; i<dimi; ++i) {
224  op(pa[i],pb[i]);
225  }
226  }
227  ++ita;
228  ++itb;
229  }
230  }
231  else {
232  std::cout << "bind ... not been here yet?\n";
233  while (ita != enda) {
234  auto pa = a.ptr(ita);
235  auto pb = b.ptr(itb);
236  for (size_t j=0; j<dimj; ++j, pa+=strideja, pb+=stridejb) {
237  auto qa = pa;
238  auto qb = pb;
239  for (size_t i=0; i<dimi; ++i, qa+=strideia, qb+=strideib) {
240  op(qa[i],qb[i]);
241  }
242  }
243  ++ita;
244  ++itb;
245  }
246  }
247  assert(itb == endb); // should be true if they really do conform
248  TTGUNUSED(endb);
249  }
250  }
251  else { // general but slowest iteration
252  std::cout << "in slow apply_binaryop ... what are you doing?\n";
253  auto ita = a.begin();
254  auto itb = b.begin();
255  const auto enda = a.end();
256  const auto endb = b.end();
257  while (ita != enda) {
258  op(*ita,*itb);
259  ++ita;
260  ++itb;
261  }
262  assert(itb == endb); // should be true if they really do conform
263  TTGUNUSED(endb);
