191 // With data streaming up the tree run compression
192 template <typename T, size_t K, Dimension NDIM>
193 void do_compress(const Key<NDIM>& key,
194  const typename ::detail::tree_types<T,K,NDIM>::compress_in_type& in,
195  typename ::detail::tree_types<T,K,NDIM>::compress_out_type& out) {
196  auto& child_slices = FunctionData<T,K,NDIM>::get_child_slices();
197  FunctionCompressedNode<T,K,NDIM> result(key); // The eventual result
198  auto& d = result.coeffs;
199 
200  T sumsq = 0.0;
201  { // Collect child coeffs and leaf info
202  FixedTensor<T,2*K,NDIM> s;
203  auto ins = ::mra::tuple_to_array_of_ptrs_const(in);
204  for (size_t i : range(Key<NDIM>::num_children)) {
205  s(child_slices[i]) = ins[i]->coeffs;
206  result.is_leaf[i] = ins[i]->is_leaf;
207  sumsq += ins[i]->sum; // Accumulate sumsq from child difference coeffs
208  }
209  filter<T,K,NDIM>(s,d); // Apply twoscale transformation
210  }
211 
212  // Recur up
213  if (key.level() > 0) {
214  FunctionReconstructedNode<T,K,NDIM> p(key);
215  p.coeffs = d(child_slices[0]);
216  d(child_slices[0]) = 0.0;
217  p.sum = d.sumabssq() + sumsq; // Accumulate sumsq of difference coeffs from this node and children
218  auto outs = ::mra::subtuple_to_array_of_ptrs<0,Key<NDIM>::num_children>(out);
219  outs[key.childindex()]->send(key.parent(), p);
220  }
221  else {
222  std::cout << "At root of compressed tree: total normsq is " << sumsq + d.sumabssq() << std::endl;
223  }
224 
225  // Send result to output tree
226  send<Key<NDIM>::num_children>(key,result,out);
227 }

241 template <typename T, size_t K, Dimension NDIM>
242 auto make_compress(rnodeEdge<T,K,NDIM>& in, cnodeEdge<T,K,NDIM>& out, const std::string& name = "compress") {
243 
244  constexpr size_t num_children = Key<NDIM>::num_children;
245 
246  using sendfuncT = decltype(&send_leaves_up<T,K,NDIM>);
247  using sendwrapT = WrapOp<sendfuncT, Key<NDIM>, typename ::detail::tree_types<T,K,NDIM>::compress_out_type, FunctionReconstructedNode<T,K,NDIM> >;
248  using compfuncT = decltype(&do_compress<T,K,NDIM>);
249  using compwrapT = typename ::detail::tree_types<T,K,NDIM>::template compwrap_type<compfuncT>;
250 
251  // Make names for terminals that connect child boxes
252  std::vector<std::string> outnames;
253  for (auto i : range(num_children)) {
254  outnames.push_back(std::string("child:")+int2bitstring(i,NDIM));
255  }
256  std::vector<std::string> innames=outnames;
257  outnames.push_back("output");
258 
259  auto s = std::unique_ptr<sendwrapT>(new sendwrapT(&send_leaves_up<T,K,NDIM>, "send_leaves_up", {"input"}, outnames));
260  auto c = std::unique_ptr<compwrapT>(new compwrapT(&do_compress<T,K,NDIM>, "do_compress", innames, outnames));
261 
262  in.set_out(s-> template in<0>()); // Connect input to s
263  out.set_in(s-> template out<num_children>()); // Connect s result to output
264  out.set_in(c-> template out<num_children>()); // Connect c result to output
265 
266  // Connect send_leaves_up to do_compress and recurrence for do_compress
267  for (auto i : range(num_children)) {
268  connect(i,i,s.get(),c.get()); // this via base class terminals
269  connect(i,i,c.get(),c.get());
270  }
271 
272  auto ins = std::make_tuple(s-> template in<0>());
273  auto outs = std::make_tuple(c-> template out<num_children>());
274  std::vector<std::unique_ptr<OpBase>> ops(2);
275  ops[0] = std::move(s);
276  ops[1] = std::move(c);
277 
278  return make_composite_op(std::move(ops), ins, outs, name);
279 }

281 template <typename T, size_t K, Dimension NDIM>
282 void do_reconstruct(const Key<NDIM>& key,
283  const std::tuple<FunctionCompressedNode<T,K,NDIM>&,FixedTensor<T,K,NDIM>&>& t,
284  std::tuple<Out<Key<NDIM>,FixedTensor<T,K,NDIM>>,rnodeOut<T,K,NDIM>>& out) {
285  const auto& child_slices = FunctionData<T,K,NDIM>::get_child_slices();
286  auto& node = std::get<0>(t);
287  const auto& from_parent = std::get<1>(t);
288  if (key.level() != 0) node.coeffs(child_slices[0]) = from_parent;
289 
290  FixedTensor<T,2*K,NDIM> s;
291  unfilter<T,K,NDIM>(node.coeffs, s);
292 
293  FunctionReconstructedNode<T,K,NDIM> r(key);
294  r.coeffs = T(0.0);
295  r.is_leaf = false;
296  ::send<1>(key, r, out); // Send empty interior node to result tree
297 
298  KeyChildren<NDIM> children(key);
299  for (auto it=children.begin(); it!=children.end(); ++it) {
300  const Key<NDIM> child= *it;
301  r.key = child;
302  r.coeffs = s(child_slices[it.index()]);
303  r.is_leaf = node.is_leaf[it.index()];
304  if (r.is_leaf) {
305  ::send<1>(child, r, out);
306  }
307  else {
308  ::send<0>(child, r.coeffs, out);
309  }
310  }
311 }
