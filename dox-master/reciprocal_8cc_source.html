20 /***
21 
22  ** design choice: if an Op is used in phases so that it is never
23  ** possible for the same key to arrive at the same time it is
24  ** permissible to reuse keys in op playing the same algorithmic role on
25  ** successive iterations
26  **
27  ** e.g., a reduce operation in an iterative solve --- since it is
28  ** a syncrhonization point it is not possible for multiple identical keys to
29  ** arrive at the same time
30  **
31  ** e.g., instead of key being pair(iteration,vector-index) it
32  ** suffices to just use key=vector-index.
33  **
34  ** this can fail at runtime in two ways --- multiple assignment
35  ** (runtime detection) or left over tasks (runtime detection) or
36  ** wrong numbers
37  **
38  ** This is essential to avoid the key potentially becoming the
39  ** union of all keys within an algorithm. Example here is
40  ** iteration index being joined to summation index in inner loop.
41 
42  ** issue: x logically depends only on i, but needs to be forwarded to all iterations k --- must we do this explicitly?
43 
44  ** easiest is to send x along with y ... change made
45 
46  ** process ... make this logical flow ... iterate until consistent and complete
47 
48  variable names here are essentially edges
49 
50  start() ---> x(i) for i n [0,N)
51 
52  guess(x(i)) ---> xy[0](i)
53 
54  iteration(xy[k](i)) --> xynew[k][i]
55 
56  residual(xy[k][i], xynew[k][i]) --> errsq[i]
57 
58  binarysum(errsq[i],k) --> errsqsum[k]
59 
60  convergence_test[k](errsqsum[k],xynew[i]) --> sends xynew[k][i] to either result[i] or xy[k+1][i]
61 
62  ***/

77 template <typename sumresultkeyT, typename valueT>
78 auto make_binary_reduce(Edge<size_t,valueT>& in, Edge<sumresultkeyT,valueT>& result, size_t N, const sumresultkeyT& result_key) {
79  using iNT = std::pair<size_t,size_t>;
80  using outT = std::tuple<Out<iNT,valueT>,Out<iNT,valueT>,Out<sumresultkeyT,valueT>>;
81  // Handles logic of propagating value up reduction tree and enventually to output
82  auto logic = [result_key](const iNT& iN, valueT sum, outT& out) {
83  const size_t i=iN.first, N=iN.second, Nisodd=(N&0x1);
84  iNT up {i>>1,(N+Nisodd)>>1};
85  if (N == 1) ::send<2>(result_key, sum, out); // result
86  else if (i&0x1) ::send<1>(up, sum, out); // right=odd
87  else ::send<0>(up, sum, out); // left=even
88  if (N>1 && i==(N-1) && Nisodd) ::send<1>(up, valueT(0.0), out); // Missing element
89  };
90 
91  // Takes vec[i] (i in [0,N]) as input and initiates reduction
92  auto startfn = [N,logic](const size_t& i, const valueT& value, outT& out) {logic(iNT(i,N), value, out);};
93 
94  auto reducefn = [logic](const iNT& iN, const valueT& left, const valueT& right, outT& out) {logic(iN, left+right, out);};
95 
96  Edge<iNT,valueT> left, right;
97  auto start = wrap<size_t>(startfn, edges(in), edges(left,right,result), "startsum", {"in"}, {"left","right","result"});
98  auto reduce= wrap<iNT>(reducefn, edges(left,right), edges(left,right,result), "reducesum", {"left","right"}, {"left","right","result"});
99  auto ins = std::make_tuple(start-> template in<0>());
100  auto outs = std::make_tuple(reduce-> template out<2>());
101  std::vector<std::unique_ptr<OpBase>> ops(2);
102  ops[0] = std::move(start);
103  ops[1] = std::move(reduce);
104  return make_composite_op(std::move(ops), ins, outs, "reduce");
105 }

132  auto f = [](const iT& i, const xT& x, std::tuple<Out<kiT,xyT>>& out) {

151 auto make_residual(Edge<kiT,xyT>&& xy, Edge<kiT,xyT>& xynew, Edge<sumreskeyT,kT>& k, Edge<iT,yT>& errsq) {
152  auto f = [](const kiT& ki, const xyT& xy, const xyT& xynew, std::tuple<Out<sumreskeyT,kT>,Out<iT,yT>>& out) {
153  yT err = (xy.second - xynew.second);
154  const kT k = ki.first;
155  const iT i = ki.second;

196  auto f = [](const kiT& ki, const bool& converged, const xyT& xynew, std::tuple<Out<kiT,xyT>,Out<iT,xyT>>& out) {
