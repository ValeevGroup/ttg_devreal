149 template <std::size_t Rank>
150 struct Key : public std::array<long, Rank> {
151  static constexpr const long max_index_width = 21;
152  static constexpr const long max_index = 1 << 21;
153  static constexpr const long max_index_square = max_index * max_index;
154  Key() = default;
155  template <typename Integer>
156  Key(std::initializer_list<Integer> ilist) {
157  std::copy(ilist.begin(), ilist.end(), this->begin());
158  assert(valid());
159  }
160  Key(std::size_t hash) {
161  static_assert(Rank == 2 || Rank == 3, "Key<Rank>::Key(hash) only implemented for Rank={2,3}");
162  if (Rank == 2) {
163  (*this)[0] = hash / max_index;
164  (*this)[1] = hash % max_index;
165  } else if (Rank == 3) {
166  (*this)[0] = hash / max_index_square;
167  (*this)[1] = (hash % max_index_square) / max_index;
168  (*this)[2] = hash % max_index;
169  }
170  }
171  std::size_t hash() const {
172  static_assert(Rank == 2 || Rank == 3, "Key<Rank>::hash only implemented for Rank={2,3}");
173  return Rank == 2 ? (*this)[0] * max_index + (*this)[1]
174  : ((*this)[0] * max_index + (*this)[1]) * max_index + (*this)[2];
175  }
176 
177  private:
178  bool valid() {
179  bool result = true;
180  for (auto &idx : *this) {
181  result = result && (idx < max_index);
182  }
183  return result;
184  }
185 };

343  for (auto i = 0ul; i != a_rowidx_to_colidx_.size(); ++i) {

368  void op(const Key<3> &key, typename baseT::input_values_tuple_type &&_ijk,
369  std::tuple<Out<Key<2>, Blk>, Out<Key<3>, Blk>> &result) {
370  const auto i = key[0];
371  const auto j = key[1];
372  const auto k = key[2];
373  long next_k;
374  bool have_next_k;
375  std::tie(next_k, have_next_k) = compute_next_k(i, j, k);
376  if (tracing()) {
377  ::ttg::print("Multiplying A[", i, "][", k, "] by B[", k, "][", j, "], next_k? ",
378  (have_next_k ? std::to_string(next_k) : "does not exist"));
379  }
380  // compute the contrib, pass the running total to the next flow, if needed
381  // otherwise write to the result flow
382  if (have_next_k) {
383  ::send<1>(Key<3>({i, j, next_k}),
384  gemm(std::move(baseT::template get<2>(_ijk)), baseT::template get<0>(_ijk), baseT::template get<1>(_ijk)), result);
385  } else
386  ::send<0>(Key<2>({i, j}), gemm(std::move(baseT::template get<2>(_ijk)), baseT::template get<0>(_ijk), baseT::template get<1>(_ijk)),
387  result);
388  }

394  // given {i,j} return first k such that A[i][k] and B[k][j] exist
395  std::tuple<long, bool> compute_first_k(long i, long j) {
396  auto a_iter_fence = a_rowidx_to_colidx_[i].end();
397  auto a_iter = a_rowidx_to_colidx_[i].begin();
398  if (a_iter == a_iter_fence) return std::make_tuple(-1, false);
399  auto b_iter_fence = b_colidx_to_rowidx_[j].end();
400  auto b_iter = b_colidx_to_rowidx_[j].begin();
401  if (b_iter == b_iter_fence) return std::make_tuple(-1, false);
402 
403  {
404  auto a_colidx = *a_iter;
405  auto b_rowidx = *b_iter;
406  while (a_colidx != b_rowidx) {
407  if (a_colidx < b_rowidx) {
408  ++a_iter;
409  if (a_iter == a_iter_fence) return std::make_tuple(-1, false);
410  a_colidx = *a_iter;
411  } else {
412  ++b_iter;
413  if (b_iter == b_iter_fence) return std::make_tuple(-1, false);
414  b_rowidx = *b_iter;
415  }
416  }
417  return std::make_tuple(a_colidx, true);
418  }
419  assert(false);
420  }

422  // given {i,j,k} such that A[i][k] and B[k][j] exist
423  // return next k such that this condition holds
424  std::tuple<long, bool> compute_next_k(long i, long j, long k) {
425  auto a_iter_fence = a_rowidx_to_colidx_[i].end();
426  auto a_iter = std::find(a_rowidx_to_colidx_[i].begin(), a_iter_fence, k);
427  assert(a_iter != a_iter_fence);
428  auto b_iter_fence = b_colidx_to_rowidx_[j].end();
429  auto b_iter = std::find(b_colidx_to_rowidx_[j].begin(), b_iter_fence, k);
430  assert(b_iter != b_iter_fence);
431  while (a_iter != a_iter_fence && b_iter != b_iter_fence) {
432  ++a_iter;
433  ++b_iter;
434  if (a_iter == a_iter_fence || b_iter == b_iter_fence) return std::make_tuple(-1, false);
435  auto a_colidx = *a_iter;
436  auto b_rowidx = *b_iter;
437  while (a_colidx != b_rowidx) {
438  if (a_colidx < b_rowidx) {
439  ++a_iter;
440  if (a_iter == a_iter_fence) return std::make_tuple(-1, false);
441  a_colidx = *a_iter;
442  } else {
443  ++b_iter;
444  if (b_iter == b_iter_fence) return std::make_tuple(-1, false);
445  b_rowidx = *b_iter;
446  }
447  }
448  return std::make_tuple(a_colidx, true);
449  }
450  abort(); // unreachable
451  }

468  std::vector<std::vector<long>> colidx_to_rowidx;

475  colidx_to_rowidx[col].push_back(row);
476  }

483  for (int k = 0; k < mat.outerSize(); ++k) { // cols, if col-major, rows otherwise
484  for (typename SpMatrix<Blk>::InnerIterator it(mat, k); it; ++it) {
485  const std::size_t row = it.row();
486  const std::size_t col = it.col();
