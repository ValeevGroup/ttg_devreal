278  size_t n = a.length();
279  int *X = new int[n*n]; // (int*) malloc(n*n*sizeof(int));
280  for(size_t i = 0; i < n; ++i) { // updating the row X[i][...]
281  for(size_t j = 0; j < n; ++j) { // updating the cell X[i][j]
282  int left_value = (j > 0 ? X[i*n + j-1] : (i+1)*get_score(a[i],'_')) +
283  get_score('_', b[j]);
284 
285  int top_value = (i > 0 ? X[(i-1)*n + j] : (j+1)*get_score('_', b[j])) +
286  get_score(a[i], '_');
287 
288  int diag_value = get_score(a[i], b[j]);
289  if (i > 0 && j > 0) { diag_value += X[(i-1)*n + j-1]; }
290  else if (i > 0) { diag_value += (i)*get_score(a[i],'_'); }
291  else if (j > 0) { diag_value += (j)*get_score('_', b[j]); }
292 
293  X[i*n+j] = std::max({left_value, top_value, diag_value});
294  //std::cout << left_value << " " << top_value << " " << diag_value << "-->" << X[i*n+j] << " ";
295  }
296  //std::cout << std::endl;
297  }

299  // returning the data at the bottom-right as the final value
300  int final_value = X[(n-1)*n+(n-1)];
