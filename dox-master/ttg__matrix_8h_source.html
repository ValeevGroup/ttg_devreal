58  static Shape add(const Shape& shape1, const Shape& shape2) {
59  if(shape1.type() != shape2.type())
60  throw std::logic_error("Shape::add(shape1,shape2): shape1.type() != shape2.type()");
61  if(shape1.type() == Type::invalid)
62  throw std::logic_error("Shape::add(shape1,shape2): shape1.type() == invalid");
63  if(!congruent(shape1, shape2))
64  throw std::logic_error("Shape::add(shape1,shape2): shape1 not congruent to shape2");
65 
66  Shape result(shape1.nrows(), shape1.ncols(), shape1.type());
67  for(std::size_t i = 0; i!=shape1.size(); ++i) {
68  const auto& shape1_row_i = shape1[i];
69  const auto& shape2_row_i = shape2[i];
70  if (shape1_row_i.empty()) {
71  if (shape2_row_i.empty()) {
72  continue; // both rows empty -> skip to next
73  }
74  else {
75  result.at(i) = shape2_row_i;
76  }
77  }
78  else {
79  if (shape2_row_i.empty()) {
80  result.at(i) = shape1_row_i;
81  }
82  else {
83  // std::merge but only keeps uniques (when shapes have norms will add the norm values)
84  // based on https://en.cppreference.com/w/cpp/iterator/inserter
85  auto first1 = shape1_row_i.begin();
86  auto last1 = shape1_row_i.end();
87  auto first2 = shape2_row_i.begin();
88  auto last2 = shape2_row_i.end();
89  auto d_first = std::inserter(result.at(i), result.at(i).end());
90  auto last_value = -1;
91  for (; first1 != last1; ++d_first) {
92  if (first2 == last2) {
93  std::copy_if(first1, last1, d_first, [last_value](auto v) { return v != last_value; });
94  break;
95  }
96  if (*first2 < *first1) {
97  if (*first2 != last_value) {
98  *d_first = *first2;
99  last_value = *first2;
100  }
101  ++first2;
102  } else {
103  if (*first1 != last_value) {
104  *d_first = *first1;
105  last_value = *first1;
106  }
107  ++first1;
108  }
109  }
110  std::copy_if(first2, last2, d_first, [last_value](auto v) { return v != last_value; });
111  }
112  }
113  }
114  return result;
115  }

146  // result[i][j] gives the j-th nonzero row for column i in matrix mat
147  template <typename T> base_t make_colidx_to_rowidx(const Eigen::SparseMatrix<T> &mat) {
148  base_t colidx_to_rowidx;
149  for (int k = 0; k < mat.outerSize(); ++k) { // cols, if col-major, rows otherwise
150  for (typename Eigen::SparseMatrix<T>::InnerIterator it(mat, k); it; ++it) {
151  const std::size_t row = it.row();
152  const std::size_t col = it.col();
153  if (col >= colidx_to_rowidx.size()) colidx_to_rowidx.resize(col + 1);
154  // in either case (col- or row-major) row index increasing for the given col
155  colidx_to_rowidx[col].push_back(row);
156  }
157  }
158  return colidx_to_rowidx;
159  }
160 
161  // result[i][j] gives the j-th nonzero column for row i in matrix mat
162  template <typename T> base_t make_rowidx_to_colidx(const Eigen::SparseMatrix<T> &mat) {
163  base_t rowidx_to_colidx;
164  for (int k = 0; k < mat.outerSize(); ++k) { // cols, if col-major, rows otherwise
165  for (typename Eigen::SparseMatrix<T>::InnerIterator it(mat, k); it; ++it) {
166  const std::size_t row = it.row();
167  const std::size_t col = it.col();
168  if (row >= rowidx_to_colidx.size()) rowidx_to_colidx.resize(row + 1);
169  // in either case (col- or row-major) col index increasing for the given row
170  rowidx_to_colidx[row].push_back(col);
171  }
172  }
173  return rowidx_to_colidx;
174  }

312  void op(typename baseT::input_values_tuple_type && ins, std::tuple<Out<Key<2>, void>> &out) {
313  const auto& shape = baseT::get<0>(ins);
314  if (shape.type() == Shape::Type::col2row) {
315  long colidx = 0;
316  for (const auto &col: shape) {
317  for (const auto rowidx: col) {
318  ::sendk<0>(Key<2>({rowidx, colidx}), out);
319  }
320  ++colidx;
321  }
322  }
323  else if (shape.type() == Shape::Type::row2col) {
324  long rowidx = 0;
325  for (const auto &row: shape) {
326  for (const auto colidx: row) {
327  ::sendk<0>(Key<2>({rowidx, colidx}), out);
328  }
329  ++rowidx;
330  }
331  }
332  else
