27 template <typename T>
28 inline BlockMatrix<T> stencil_computation(int i, int j, int M, int N, BlockMatrix<T> bm, BlockMatrix<T> left,
29  BlockMatrix<T> top, BlockMatrix<T> right, BlockMatrix<T> bottom) {
30  // i==0 -> no top block
31  // j==0 -> no left block
32  // i==M-1 -> no bottom block
33  // j==N-1 -> no right block
34  int MB = bm.rows();
35  int NB = bm.cols();
36  BlockMatrix<T> current = bm;
37  for (int ii = 0; ii < MB; ++ii) {
38  for (int jj = 0; jj < NB; ++jj) {
39  current(ii,jj) = (current(ii,jj) + ((ii == 0) ? (i > 0 ? top(MB - 1, jj) : 0.0) : current(ii - 1, jj)));
40  current(ii,jj) = (current(ii,jj) + ((ii == MB - 1) ? (i < M - 1 ? bottom(0, jj) : 0.0) : current(ii + 1, jj)));
41  current(ii,jj) = (current(ii,jj) + ((jj == 0) ? (j > 0 ? left(ii, NB - 1) : 0.0) : current(ii, jj - 1)));
42  current(ii,jj) = (current(ii,jj) + ((jj == NB - 1) ? (j < N - 1 ? right(ii, 0) : 0.0) : current(ii, jj + 1)));
43  current(ii,jj) = current(ii,jj) * 0.25;
44  }
45  }
46  return current;
47 }

52  for (int i = 0; i < n_brows; i++) {
53  for (int j = 0; j < n_bcols; j++) {
54  BlockMatrix<T> left, top, right, bottom;
55 
56  if (i < n_brows - 1) bottom = ((*m)(i + 1, j));

58  if (j > 0) left = ((*m)(i, j - 1));
59  if (i > 0) top = ((*m)(i - 1, j));
60 
61  (*result)(i,j) = stencil_computation(i, j, n_brows, n_bcols, ((*m)(i,j)), (left), (top), (right), (bottom));
62  }
